<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" layout="absolute" xmlns:diagrammer="com.anotherflexdev.diagrammer.*" backgroundGradientAlphas="[1.0, 1.0]" backgroundGradientColors="[#FFFFFF, #FFFFFF]" creationComplete="init()" viewSourceURL="srcview/index.html">

    <mx:Script>
        <![CDATA[
            import diagrammer.SendNotification;
            import diagrammer.SamplesInProcess;
            import diagrammer.WaitingCustomerConfirmation;
            import com.anotherflexdev.diagrammer.Link;
            import com.anotherflexdev.diagrammer.BaseNode;
            import mx.containers.Canvas;
            import diagrammer.QuotationInProcess;
            import diagrammer.WaitingInitialPayment;
            
            private function init():void {
                var link:Link = new Link;
                link.fromNode = new QuotationInProcess;
                link.fromNode.x = 100;
                link.fromNode.y = 100;
                link.toNode = new WaitingCustomerConfirmation;
                link.toNode.x = 400;
                link.toNode.y = 91;
                link.linkName = "After Research";
                link.fromNode.addLeavingLink(link);
                link.toNode.addArrivingLink(link);
                this.diagram.addChild(link.fromNode);
                this.diagram.addChild(link.toNode);
                this.diagram.addChildAt(link, 0);
            }
            
            private function addNode(clazz:Class):void{
                var node:Canvas = new clazz;
                node.x = 100;
                node.y = 100;
                diagram.addChild(node);
            }
        ]]>
    </mx:Script>
